# Code

`Code` is a typography component for rendering inline code with monospace styling.

## Import

```tsx
import { Code } from 'radix-native-ui';
```

## Usage

```tsx
import { Text, Code } from 'radix-native-ui';

<Text>
  Use the <Code>useState</Code> hook for state management.
</Text>
```

## Props

<PropTable>
  <Prop name="children" type="ReactNode">
    Code content.
  </Prop>
  <Prop name="style" type="StyleProp<TextStyle>">
    Custom styles.
  </Prop>
  <Prop name="size" type="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 | 9" default="2">
    Font size from theme scale.
  </Prop>
  <Prop name="color" type="string">
    Text color.
  </Prop>
  <Prop name="backgroundColor" type="string">
    Background color.
  </Prop>
</PropTable>
