# Switch

`Switch` is a form component for toggling between on/off states.

## Import

```tsx
import { Switch } from 'radix-native-ui';
```

## Usage

```tsx
import { Switch } from 'radix-native-ui';

<Switch
  checked={enabled}
  onCheckedChange={setEnabled}
/>
```

## Props

<PropTable>
  <Prop name="defaultChecked" type="boolean" default="false">
    Whether the switch is checked by default (uncontrolled mode).
  </Prop>
  <Prop name="checked" type="boolean">
    Whether the switch is on (controlled mode).
  </Prop>
  <Prop name="onCheckedChange" type="(checked: boolean) => void">
    Called when checked state changes.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the switch is disabled.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3'" default="'2'">
    Size of the switch.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'surface'" default="'solid'">
    Visual variant.
  </Prop>
  <Prop name="color" type="Color">
    Accent color when checked.
  </Prop>
  <Prop name="thumbColor" type="string">
    Custom color for the thumb.
  </Prop>
  <Prop name="radius" type="RadiusSize" default="'full'">
    Border radius of the switch.
  </Prop>
  <Prop name="label" type="string">
    Label text displayed next to the switch.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
  <Prop name="hapticFeedback" type="boolean" default="true">
    Enable haptic feedback when toggling.
  </Prop>
</PropTable>

### Haptic Feedback

Switches provide haptic feedback by default when toggled. You can disable this with the `hapticFeedback` prop:

```tsx
<Switch
  checked={enabled}
  onCheckedChange={setEnabled}
  hapticFeedback={false}
/>
```
