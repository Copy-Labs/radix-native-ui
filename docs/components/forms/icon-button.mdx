# IconButton

`IconButton` is a button component specifically designed for rendering icons with various styles.

## Import

```tsx
import { IconButton } from 'radix-native-ui';
```

## Usage

```tsx
import { IconButton } from 'radix-native-ui';
import { Icon } from './icons';

<IconButton onPress={() => console.log('Pressed!')}>
  <Icon name="plus" />
</IconButton>
```

## Props

<PropTable>
  <Prop name="children" type="ReactNode">
    Icon content.
  </Prop>
  <Prop name="style" type="ViewStyle">
    Custom styles.
  </Prop>
  <Prop name="variant" type="'surface' | 'solid' | 'soft' | 'outline' | 'ghost'" default="'solid'">
    Visual style variant.
  </Prop>
  <Prop name="size" type="1 | 2 | 3 | 4" default="2">
    Size of the button.
  </Prop>
  <Prop name="color" type="Color">
    Accent color.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the button is disabled.
  </Prop>
  <Prop name="loading" type="boolean" default="false">
    Shows loading spinner.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="onPress" type="(event: GestureResponderEvent) => void">
    Called when button is pressed.
  </Prop>
  <Prop name="onLongPress" type="(event: GestureResponderEvent) => void">
    Called when button is long-pressed.
  </Prop>
  <Prop name="delayLongPress" type="number" default="500">
    Duration (in ms) before onLongPress is called.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label for screen readers.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'">
    Border radius override.
  </Prop>
  <Prop name="asChild" type="boolean" default="false">
    Merge props onto child element.
  </Prop>
  <Prop name="hapticFeedback" type="boolean" default="true">
    Enable haptic feedback on press.
  </Prop>
</PropTable>

### Haptic Feedback

Icon buttons provide haptic feedback by default when pressed. You can disable this with the `hapticFeedback` prop:

```tsx
<IconButton hapticFeedback={false} accessibilityLabel="No haptic">
  <Icon name="settings" />
</IconButton>
```
