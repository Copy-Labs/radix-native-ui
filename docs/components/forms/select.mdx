# Select

`Select` displays a list of options for the user to pick from—triggered by a button.

## Import

```tsx
import { Select } from 'radix-native-ui';
```

## Anatomy

The Select component is composed of the following parts:

```tsx
<Select.Root>
  <Select.Trigger>
    <Select.Value />
  </Select.Trigger>
  <Select.Portal>
    <Select.Overlay />
    <Select.Content>
      <Select.Group>
        <Select.Label />
        <Select.Item />
      </Select.Group>
      <Select.Separator />
    </Select.Content>
  </Select.Portal>
</Select.Root>
```

## Props

### Root

<PropTable>
  <Prop name="value" type="string">
    Currently selected value (controlled mode).
  </Prop>
  <Prop name="defaultValue" type="string">
    Default value (uncontrolled mode).
  </Prop>
  <Prop name="onValueChange" type="(value: string) => void">
    Called when selection changes.
  </Prop>
  <Prop name="open" type="boolean">
    Whether the select is open (controlled mode).
  </Prop>
  <Prop name="defaultOpen" type="boolean" default="false">
    Whether the select is open by default (uncontrolled mode).
  </Prop>
  <Prop name="onOpenChange" type="(open: boolean) => void">
    Called when open state changes.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the select is disabled.
  </Prop>
  <Prop name="color" type="Color" default="theme.accentColor">
    Accent color for selected items.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'outline' | 'surface' | 'ghost'" default="'soft'">
    Visual style variant for the selected value.
  </Prop>
</PropTable>

### Trigger

<PropTable>
  <Prop name="asChild" type="boolean" default="false">
    When true, the trigger will clone its child and inject the ref and onPress handler. When false, wraps children in an AnimatedPressable.
  </Prop>
  <Prop name="children" type="ReactNode" required>
    The trigger content. Usually a Button or custom pressable component.
  </Prop>
</PropTable>

### Value

<PropTable>
  <Prop name="placeholder" type="string" default="'Select an option'">
    Placeholder text when no option is selected.
  </Prop>
  <Prop name="style" type="TextStyle">
    Custom styles for the value text.
  </Prop>
</PropTable>

### Content

<PropTable>
  <Prop name="side" type="'top' | 'right' | 'bottom' | 'left'" default="'bottom'">
    Preferred side of the trigger to render on.
  </Prop>
  <Prop name="sideOffset" type="number" default="4">
    Distance in pixels from the trigger.
  </Prop>
  <Prop name="align" type="'start' | 'center' | 'end'" default="'start'">
    Preferred alignment against the trigger.
  </Prop>
  <Prop name="alignOffset" type="number" default="0">
    Offset in pixels from the alignment.
  </Prop>
  <Prop name="avoidCollisions" type="boolean" default="true">
    Whether to avoid collisions with the screen edges.
  </Prop>
  <Prop name="size" type="1 | 2 | 3 | 4" default="2">
    Size of the content and items.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles for the content container.
  </Prop>
</PropTable>

### Item

<PropTable>
  <Prop name="value" type="string" required>
    The unique value of the item.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the item is disabled.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles for the item.
  </Prop>
  <Prop name="children" type="ReactNode" required>
    The item label content.
  </Prop>
</PropTable>

### Group

<PropTable>
  <Prop name="children" type="ReactNode" required>
    Group content - usually contains Label and Item components.
  </Prop>
</PropTable>

### Label

<PropTable>
  <Prop name="children" type="ReactNode" required>
    The label text.
  </Prop>
  <Prop name="style" type="TextStyle">
    Custom styles for the label.
  </Prop>
</PropTable>

### Separator

<PropTable>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles for the separator line.
  </Prop>
</PropTable>

## Examples

### Basic Select

```tsx
import { Select, Button } from 'radix-native-ui';
import { useState } from 'react';

function BasicSelect() {
  const [fruit, setFruit] = useState<string>('');

  return (
    <Select.Root value={fruit} onValueChange={setFruit}>
      <Select.Trigger asChild>
        <Button variant="solid">
          <Button.Label>
            <Select.Value placeholder="Select a fruit" />
          </Button.Label>
        </Button>
      </Select.Trigger>
      <Select.Portal>
        <Select.Overlay />
        <Select.Content>
          <Select.Item value="apple">Apple</Select.Item>
          <Select.Item value="banana">Banana</Select.Item>
          <Select.Item value="orange">Orange</Select.Item>
          <Select.Item value="grape">Grape</Select.Item>
        </Select.Content>
      </Select.Portal>
    </Select.Root>
  );
}
```

## Animation & Haptics

The Select component uses `AnimatedPressable` for enhanced interaction feedback:

- **Trigger**: Provides press animation and haptic feedback when opening/closing the dropdown
- **Items**: Each item has press animation and haptic feedback when selected

## Accessibility

- Supports screen readers with proper `accessibilityRole="menuitem"` on items
- Disabled items announce as disabled via `accessibilityState`
- Selected state is properly announced
- Keyboard navigation support through modal presentation
