# TextField

`TextField` is a form component for capturing single-line text input.

## Import

```tsx
import { TextField } from 'radix-native-ui';
```

## Usage

```tsx
import { TextField } from 'radix-native-ui';

<TextField
  placeholder="Enter your name"
  value={name}
  onChangeText={setName}
/>
```

## Props

<PropTable>
  <Prop name="value" type="string">
    Current input value.
  </Prop>
  <Prop name="onChangeText" type="(text: string) => void">
    Called when text changes.
  </Prop>
  <Prop name="placeholder" type="string">
    Placeholder text.
  </Prop>
  <Prop name="variant" type="'surface' | 'soft' | 'outline'" default="'outline'">
    Visual style variant.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3'" default="'2'">
    Size of the input.
  </Prop>
  <Prop name="color" type="Color">
    Accent color for focus state.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the input is disabled.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'">
    Border radius override.
  </Prop>
  <Prop name="label" type="string">
    Label text displayed above the input.
  </Prop>
  <Prop name="error" type="string">
    Error message displayed below the input.
  </Prop>
  <Prop name="multiline" type="boolean" default="false">
    Whether this is a multiline input.
  </Prop>
  <Prop name="secureTextEntry" type="boolean" default="false">
    Masks text for passwords.
  </Prop>
  <Prop name="keyboardType" type="KeyboardTypeOptions" default="'default'">
    Keyboard type.
  </Prop>
  <Prop name="autoCapitalize" type="'none' | 'sentences' | 'words' | 'characters'">
    Auto capitalization.
  </Prop>
  <Prop name="autoCorrect" type="boolean">
    Auto correct enabled.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
</PropTable>

## Examples

### Basic Input

```tsx
<TextField
  placeholder="Enter text"
  value={value}
  onChangeText={setValue}
/>
```

### Variants

```tsx
<TextField variant="outline" placeholder="Outline style" />
<TextField variant="surface" placeholder="Surface style" />
<TextField variant="soft" placeholder="Soft style" />
```

### Sizes

```tsx
<TextField size="1" placeholder="Small" />
<TextField size="2" placeholder="Medium (default)" />
<TextField size="3" placeholder="Large" />
```

### Email Input

```tsx
<TextField
  placeholder="Enter your email"
  value={email}
  onChangeText={setEmail}
  keyboardType="email-address"
  autoCapitalize="none"
  autoCorrect={false}
/>
```

### Password Input

```tsx
<TextField
  placeholder="Enter password"
  value={password}
  onChangeText={setPassword}
  secureTextEntry
/>
```

### With Error

```tsx
<TextField
  placeholder="Enter email"
  value={email}
  onChangeText={setEmail}
  error="Please enter a valid email"
/>
```

### With Slots

TextField supports slots for adding icons or buttons on the left or right side of the input:

```tsx
import { TextField } from 'radix-native-ui';
import { SearchIcon, CrossIcon } from './icons';

// Left slot (auto-detected)
<TextField.Root placeholder="Search...">
  <TextField.Slot>
    <SearchIcon />
  </TextField.Slot>
</TextField.Root>

// Right slot (explicit)
<TextField.Root placeholder="Clear">
  <TextField.Slot side="right">
    <CrossIcon />
  </TextField.Slot>
</TextField.Root>

// Both sides
<TextField.Root placeholder="Search...">
  <TextField.Slot side="left">
    <SearchIcon />
  </TextField.Slot>
  <TextField.Slot side="right">
    <CrossIcon />
  </TextField.Slot>
</TextField.Root>
```

### Disabled

```tsx
<TextField
  placeholder="Disabled input"
  disabled
/>
```
