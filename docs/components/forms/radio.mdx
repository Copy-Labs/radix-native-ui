# Radio

`Radio` is a form component for capturing single selections within a group.

## Import

```tsx
import { Radio } from 'radix-native-ui';
```

## Usage

```tsx
import { Radio } from 'radix-native-ui';

// With label prop
<Radio value="option1" label="Option 1" />

// With children (custom content)
<Radio value="option1">
  <Text>Custom content</Text>
</Radio>

// Controlled
<Radio value="option1" selected={isSelected} onSelect={handleSelect} />

// Uncontrolled (with defaultChecked)
<Radio value="option1" defaultChecked />
```

## Props

<PropTable>
  <Prop name="defaultChecked" type="boolean" default="false">
    Whether the radio is checked by default (uncontrolled mode).
  </Prop>
  <Prop name="value" type="string" required>
    Value for this radio option.
  </Prop>
  <Prop name="selected" type="boolean">
    Whether the radio is selected (controlled mode).
  </Prop>
  <Prop name="onSelect" type="(value: string) => void">
    Callback when radio is selected.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether this radio is disabled.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3'" default="'2'">
    Size of the radio.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'surface' | 'outline'" default="'solid'">
    Visual style variant.
  </Prop>
  <Prop name="color" type="Color">
    Accent color.
  </Prop>
  <Prop name="label" type="string">
    Label text displayed next to the radio.
  </Prop>
  <Prop name="children" type="ReactNode">
    Custom content displayed next to the radio (overrides label).
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
  <Prop name="hapticFeedback" type="boolean" default="true">
    Enable haptic feedback when selected.
  </Prop>
</PropTable>

### Haptic Feedback

Radio buttons provide haptic feedback by default when selected. You can disable this with the `hapticFeedback` prop:

```tsx
<Radio value="1" label="No haptic" hapticFeedback={false} />
<Radio value="2" label="With haptic" />
```

## Examples

### Basic Radio

```tsx
<Radio value="apple" label="Apple" />
<Radio value="banana" label="Banana" />
<Radio value="orange" label="Orange" />
```

### Sizes

```tsx
<Radio value="1" size="1" label="Small" />
<Radio value="2" size="2" label="Medium" />
<Radio value="3" size="3" label="Large" />
```

### With Color

```tsx
<Radio value="1" color="blue" label="Option 1" />
<Radio value="2" color="blue" label="Option 2" />
```

### Disabled Option

```tsx
<Radio value="1" label="Available" />
<Radio value="2" label="Unavailable" disabled />
```

### Variants

```tsx
<Radio value="1" variant="solid" label="Solid" defaultChecked />
<Radio value="2" variant="soft" label="Soft" defaultChecked />
<Radio value="3" variant="surface" label="Surface" defaultChecked />
<Radio value="4" variant="outline" label="Outline" defaultChecked />
```
