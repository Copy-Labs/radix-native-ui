# TextArea

`TextArea` is a form component for capturing multi-line text input.

## Import

```tsx
import { TextArea } from 'radix-native-ui';
```

## Usage

```tsx
import { TextArea } from 'radix-native-ui';

<TextArea
  placeholder="Enter your message"
  value={message}
  onChangeText={setMessage}
/>
```

## Props

<PropTable>
  <Prop name="value" type="string">
    Current input value.
  </Prop>
  <Prop name="onChangeText" type="(text: string) => void">
    Called when text changes.
  </Prop>
  <Prop name="placeholder" type="string">
    Placeholder text.
  </Prop>
  <Prop name="variant" type="'surface' | 'soft' | 'outline'" default="'outline'">
    Visual style variant.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3'" default="'2'">
    Size of the input.
  </Prop>
  <Prop name="color" type="Color">
    Accent color for focus state.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the input is disabled.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'">
    Border radius override.
  </Prop>
  <Prop name="label" type="string">
    Label text displayed above the input.
  </Prop>
  <Prop name="error" type="string">
    Error message displayed below the input.
  </Prop>
  <Prop name="rows" type="number" default="4">
    Number of visible lines.
  </Prop>
  <Prop name="autoGrow" type="boolean" default="false">
    Auto-grow height.
  </Prop>
  <Prop name="numberOfLines" type="number" default="4">
    Number of visible lines.
  </Prop>
  <Prop name="maxLength" type="number">
    Maximum character length.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
</PropTable>

## Examples

### Basic TextArea

```tsx
<TextArea
  placeholder="Enter your message"
  value={message}
  onChangeText={setMessage}
/>
```

### With Character Limit

```tsx
<TextArea
  placeholder="Write a bio (max 200 characters)"
  value={bio}
  onChangeText={setBio}
  maxLength={200}
/>
```

### Different Sizes

```tsx
<TextArea size="1" placeholder="Small" />
<TextArea size="2" placeholder="Medium (default)" />
<TextArea size="3" placeholder="Large" />
```

### With Error

```tsx
<TextArea
  placeholder="Enter description"
  value={description}
  onChangeText={setDescription}
  error="Description is required"
/>
```

### Contact Form

```tsx
<Flex direction="column" gap={3}>
  <TextField placeholder="Your Name" />
  <TextField placeholder="Email" keyboardType="email-address" />
  <TextArea
    placeholder="Your Message"
    rows={6}
  />
  <Button>Send Message</Button>
</Flex>
```

### With Slots

TextArea supports slots for adding icons or buttons on the left or right side of the input:

```tsx
import { TextArea } from 'radix-native-ui';
import { SearchIcon } from './icons';

// Left slot
<TextArea.Root placeholder="Write a comment...">
  <TextArea.Slot side="left">
    <SearchIcon />
  </TextArea.Slot>
</TextArea.Root>
```
