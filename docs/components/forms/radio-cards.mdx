# RadioCards

`RadioCards` provides a card-style interface for selecting a single option from a group.

## Import

```tsx
import { RadioCards } from 'radix-native-ui';
```

## Usage

```tsx
import { RadioCards } from 'radix-native-ui';

<RadioCards.Root value={value} onValueChange={setValue}>
  <RadioCards.Item value="1">
    <Flex direction="column">
      <Text weight="medium">Option 1</Text>
      <Text size={2} color="gray">Description here</Text>
    </Flex>
  </RadioCards.Item>
  <RadioCards.Item value="2">
    <Flex direction="column">
      <Text weight="medium">Option 2</Text>
      <Text size={2} color="gray">Description here</Text>
    </Flex>
  </RadioCards.Item>
</RadioCards.Root>
```

## Props

### RadioCards.Root

<PropTable>
  <Prop name="children" type="ReactNode">
    RadioCard items.
  </Prop>
  <Prop name="value" type="string">
    Currently selected value.
  </Prop>
  <Prop name="defaultValue" type="string">
    Default selected value (uncontrolled).
  </Prop>
  <Prop name="onValueChange" type="(value: string) => void">
    Called when selection changes.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'surface' | 'outline'" default="'surface'">
    Visual variant for all cards.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3'" default="'2'">
    Size of the cards.
  </Prop>
  <Prop name="color" type="Color">
    Accent color.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="radius" type="RadiusSize" default="'medium'">
    Border radius of the cards.
  </Prop>
  <Prop name="direction" type="'row' | 'column'" default="'column'">
    Layout direction of the cards.
  </Prop>
  <Prop name="gap" type="number">
    Gap between cards.
  </Prop>
  <Prop name="side" type="'left' | 'right'" default="'right'">
    Indicator position.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether all items are disabled.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles for the container.
  </Prop>
</PropTable>

### RadioCards.Item

<PropTable>
  <Prop name="children" type="ReactNode">
    Card content.
  </Prop>
  <Prop name="value" type="string" required>
    Unique value for this item.
  </Prop>
  <Prop name="side" type="'left' | 'right'">
    Indicator position (overrides group setting).
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether this item is disabled.
  </Prop>
  <Prop name="hapticFeedback" type="boolean" default="true">
    Enable haptic feedback when selected.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles for this card.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
</PropTable>

### Animation

Radio card items include subtle press animations for a tactile feel:

- **Press Animation**: Cards scale down slightly (0.975x) when pressed
