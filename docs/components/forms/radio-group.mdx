# RadioGroup

`RadioGroup` is a component for managing a group of radio buttons where only one can be selected. It uses the compound component pattern for better flexibility and consistency with other components.

## Import

```tsx
import { RadioGroup } from 'radix-native-ui';
```

## Anatomy

```tsx
<RadioGroup.Root>
  <RadioGroup.Item value="option1">Option 1</RadioGroup.Item>
  <RadioGroup.Item value="option2">Option 2</RadioGroup.Item>
</RadioGroup.Root>
```

## Props

### Root

<PropTable>
  <Prop name="value" type="string">
    Currently selected value (controlled mode).
  </Prop>
  <Prop name="defaultValue" type="string">
    Default selected value (uncontrolled mode).
  </Prop>
  <Prop name="onValueChange" type="(value: string) => void">
    Called when selection changes.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether all radios are disabled.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3'" default="'2'">
    Size of all radios.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'surface' | 'outline'" default="'soft'">
    Visual variant for all radios.
  </Prop>
  <Prop name="color" type="Color">
    Accent color for all radios.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="direction" type="'row' | 'column'" default="'column'">
    Direction of radio items.
  </Prop>
  <Prop name="gap" type="number">
    Gap between radio items.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles for the container.
  </Prop>
</PropTable>

### Item

<PropTable>
  <Prop name="value" type="string" required>
    Unique value for this radio item.
  </Prop>
  <Prop name="label" type="string">
    Label text displayed next to the radio.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether this specific item is disabled.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'surface' | 'outline'">
    Visual variant (overrides group variant).
  </Prop>
  <Prop name="color" type="Color">
    Custom color (overrides group color).
  </Prop>
  <Prop name="highContrast" type="boolean">
    High contrast mode (overrides group setting).
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
</PropTable>

## Examples

### Basic Group

```tsx
<RadioGroup.Root value={value} onValueChange={setValue}>
  <RadioGroup.Item value="apple" label="Apple" />
  <RadioGroup.Item value="banana" label="Banana" />
  <RadioGroup.Item value="orange" label="Orange" />
</RadioGroup.Root>
```

### With Label

```tsx
<Flex direction="column" gap={2}>
  <Text weight="medium">Select your favorite fruit:</Text>
  <RadioGroup.Root value={value} onValueChange={setValue}>
    <RadioGroup.Item value="apple" label="Apple" />
    <RadioGroup.Item value="banana" label="Banana" />
    <RadioGroup.Item value="orange" label="Orange" />
  </RadioGroup.Root>
</Flex>
```

### Custom Content

```tsx
<RadioGroup.Root value={method} onValueChange={setMethod}>
  <RadioGroup.Item value="card">
    <Flex direction="row" gap={2} align="center">
      <Icon name="credit-card" />
      <Text>Credit Card</Text>
    </Flex>
  </RadioGroup.Item>
  <RadioGroup.Item value="paypal">
    <Flex direction="row" gap={2} align="center">
      <Icon name="paypal" />
      <Text>PayPal</Text>
    </Flex>
  </RadioGroup.Item>
</RadioGroup.Root>
```

### Disabled Group

```tsx
<RadioGroup.Root disabled defaultValue="1">
  <RadioGroup.Item value="1" label="Option 1" />
  <RadioGroup.Item value="2" label="Option 2" />
</RadioGroup.Root>
```

### Disabled Item

```tsx
<RadioGroup.Root value={value} onValueChange={setValue}>
  <RadioGroup.Item value="1" label="Option 1" />
  <RadioGroup.Item value="2" label="Option 2" disabled />
  <RadioGroup.Item value="3" label="Option 3" />
</RadioGroup.Root>
```

### With Styling

```tsx
<RadioGroup.Root
  value={value}
  onValueChange={setValue}
  color="indigo"
  variant="surface"
  size="3"
  gap={12}
>
  <RadioGroup.Item value="1" label="Surface Indigo" />
  <RadioGroup.Item value="2" label="Another Option" />
</RadioGroup.Root>
```
