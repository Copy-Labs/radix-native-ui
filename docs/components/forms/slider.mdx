# Slider

`Slider` is a form component for selecting numeric values within a range.

## Import

```tsx
import { Slider } from 'radix-native-ui';
```

## Usage

```tsx
import { Slider } from 'radix-native-ui';

<Slider
  value={[volume]}
  onValueChange={([v]) => setVolume(v)}
  min={0}
  max={100}
  step={1}
/>
```

## Props

<PropTable>
  <Prop name="value" type="number | [number, number]">
    Current value(s) - single or range.
  </Prop>
  <Prop name="defaultValue" type="number | [number, number]">
    Default value (uncontrolled).
  </Prop>
  <Prop name="onValueChange" type="(value: number | [number, number]) => void">
    Called when value changes.
  </Prop>
  <Prop name="onValueChangeEnd" type="(value: number | [number, number]) => void">
    Called when value change ends (drag end).
  </Prop>
  <Prop name="min" type="number" default="0">
    Minimum value.
  </Prop>
  <Prop name="max" type="number" default="100">
    Maximum value.
  </Prop>
  <Prop name="step" type="number" default="1">
    Step increment.
  </Prop>
  <Prop name="minStepsBetweenThumbs" type="number" default="0">
    Minimum steps between thumbs (for range slider).
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the slider is disabled.
  </Prop>
  <Prop name="size" type="'1' | '2' | '3' | '4'" default="'2'">
    Size of the slider.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'surface' | 'outline'" default="'solid'">
    Visual variant.
  </Prop>
  <Prop name="color" type="Color">
    Accent color.
  </Prop>
  <Prop name="radius" type="RadiusSize" default="'full'">
    Border radius.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="label" type="string">
    Label to display above the slider.
  </Prop>
  <Prop name="showValueLabel" type="boolean" default="false">
    Whether to show value label.
  </Prop>
  <Prop name="valueLabelFormatter" type="(value: number) => string">
    Custom formatter for value label.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
  <Prop name="accessibilityHint" type="string">
    Accessibility hint.
  </Prop>
  <Prop name="style" type="ViewStyle">
    Custom styles.
  </Prop>
  <Prop name="hapticFeedback" type="boolean" default="true">
    Enable haptic feedback during drag.
  </Prop>
</PropTable>

### Haptic Feedback

Sliders provide haptic feedback by default when dragging starts, ends, and when tapping on the track. You can disable this with the `hapticFeedback` prop:

```tsx
<Slider
  value={[value]}
  onValueChange={([v]) => setValue(v)}
  hapticFeedback={false}
/>
```

## Examples

### Basic Slider

```tsx
<Slider
  value={[value]}
  onValueChange={([v]) => setValue(v)}
/>
```

### With Range

```tsx
<Slider
  value={[min, max]}
  onValueChange={([min, max]) => setRange(min, max)}
  min={0}
  max={1000}
  step={10}
/>
```

### Volume Control

```tsx
<Flex direction="column" gap={2}>
  <Flex justify="between">
    <Text>Volume</Text>
    <Text>{volume}%</Text>
  </Flex>
  <Slider
    value={[volume]}
    onValueChange={([v]) => setVolume(v)}
    color="blue"
  />
</Flex>
```

### Price Range Filter

```tsx
<Card>
  <Flex direction="column" gap={3}>
    <Heading size={3}>Price Range</Heading>
    <Flex justify="between">
      <Text>${priceRange[0]}</Text>
      <Text>${priceRange[1]}</Text>
    </Flex>
    <Slider
      value={priceRange}
      onValueChange={setPriceRange}
      min={0}
      max={1000}
      step={50}
      minStepsBetweenThumbs={1}
    />
  </Flex>
</Card>
```
