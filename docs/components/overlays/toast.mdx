# Toast

`Toast` is an overlay component for displaying brief, non-critical notifications that appear temporarily. Toast uses a hook-based pattern combined with `ThemeProvider`.

## Import

```tsx
import { Toast, useToast } from 'radix-native-ui';
```

## Setup

Wrap your app with `ThemeProvider` and configure the toast settings:

```tsx
import { ThemeProvider } from 'radix-native-ui';

<ThemeProvider
  toastConfig={{
    position: 'bottom',
    maxVisible: 3,
    duration: 5000,
    swipeToDismiss: true,
  }}
>
  <App />
</ThemeProvider>
```

## Usage

```tsx
import { Toast, Button, useToast } from 'radix-native-ui';

function App() {
  const toast = useToast();

  return (
    <Button onPress={() => toast.show({ title: 'Notification', description: 'Your changes have been saved.' })}>
      Show Toast
    </Button>
  );
}
```

## Components

### Toast.Viewport

Container where toasts are rendered. Must be placed in your app layout.

### Toast.Title

Title text for custom toast content.

### Toast.Description

Description text for custom toast content.

### Toast.Close

Close button for custom toast content.

### Toast.Action

Action button for custom toast content.

## Hook: useToast

The `useToast` hook returns a toast controller with methods to show and manage toasts:

```tsx
const toast = useToast();

// Show a toast with options
toast.show({ title: 'Message' });

// Shorthand methods
toast.success({ title: 'Success!' });
toast.error({ title: 'Error!' });
toast.warning({ title: 'Warning!' });
toast.info({ title: 'Info' });
toast.accent({ title: 'Accent' });

// Programmatic control
toast.hide(id);
toast.hideAll();
toast.update(id, { title: 'Updated' });
```

  <Prop name="show" type="(options: ToastOptions) => string">

  </Prop>
  <Prop name="error" type="(options: ToastOptions) => string">

  </Prop>
  <Prop name="warning" type="(options: ToastOptions) => string">

  </Prop>
  <Prop name="info" type="(options: ToastOptions) => string">

  </Prop>
  <Prop name="accent" type="(options: ToastOptions) => string">

  </Prop>
  <Prop name="hide" type="(id: string) => void">

## ToastOptions

<PropTable>
  <Prop name="title" type="string" required>
    Toast title (required).
  </Prop>
  <Prop name="description" type="string">
    Optional description text.
  </Prop>
  <Prop name="variant" type="'default' | 'success' | 'error' | 'warning' | 'info' | 'accent'" default="'default'">
    Visual variant.
  </Prop>
  <Prop name="color" type="Color">
    Override color (overrides variant color).
  </Prop>
  <Prop name="duration" type="number">
    Duration in milliseconds before auto-dismiss. Set to Infinity for no auto-dismiss.
  </Prop>
  <Prop name="position" type="'top' | 'bottom'">
    Position override for this specific toast.
  </Prop>
  <Prop name="action" type="{ label: string; onPress: () => void }">
    Optional action button.
  </Prop>
  <Prop name="onDismiss" type="() => void">
    Callback when toast is dismissed.
  </Prop>
  <Prop name="onAutoClose" type="() => void">
    Callback when toast auto-closes.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label for screen readers.
  </Prop>
  <Prop name="icon" type="ReactNode">
    Custom icon.
  </Prop>
  <Prop name="showClose" type="boolean" default="true">
    Whether to show close button.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom style for the toast container.
  </Prop>
</PropTable>

## ThemeProvider: toastConfig

Configure toast behavior via `ThemeProvider`:

<PropTable>
  <Prop name="position" type="'top' | 'bottom'" default="'bottom'">
    Position of toast container.
  </Prop>
  <Prop name="maxVisible" type="number" default="3">
    Maximum number of visible toasts in stack.
  </Prop>
  <Prop name="duration" type="number" default="5000">
    Default duration in milliseconds.
  </Prop>
  <Prop name="defaultVariant" type="'default' | 'success' | 'error' | 'warning' | 'info' | 'accent'" default="'default'">
    Default variant.
  </Prop>
  <Prop name="swipeToDismiss" type="boolean" default="true">
    Whether toasts can be dismissed by swiping.
  </Prop>
  <Prop name="gap" type="number" default="8">
    Gap between stacked toasts (in pixels).
  </Prop>
  <Prop name="hapticFeedback" type="boolean" default="true">
    Whether to enable haptic feedback.
  </Prop>
</PropTable>

## Examples

### Basic Toast

```tsx
function ToastDemo() {
  const toast = useToast();

  return (
    <Button onPress={() => toast.show({ title: 'Default toast message' })}>
      Show Default Toast
    </Button>
  );
}
```

### Success Toast

```tsx
<Button color="green" onPress={() => toast.success({ title: 'Operation successful!' })}>
  Show Success Toast
</Button>
```

### Error Toast

```tsx
<Button color="red" onPress={() => toast.error({ title: 'Something went wrong' })}>
  Show Error Toast
</Button>
```

### Toast with Description

```tsx
toast.success({
  title: 'File uploaded',
  description: 'Your document has been successfully uploaded to the server.',
});
```

### Toast with Action

```tsx
toast.error({
  title: 'File deleted',
  description: 'Your file has been moved to trash.',
  action: {
    label: 'Undo',
    onPress: () => toast.success({ title: 'File restored!' }),
  },
  showClose: false,
});
```

### Toast with Custom Icon

```tsx
toast.show({
  title: 'Upload complete',
  icon: <Ionicons name="checkmark-circle" size={24} color="green" />,
});
```

### Custom Duration

```tsx
// 2 second toast
toast.show({
  title: 'Quick message',
  duration: 2000,
});

// Persistent toast (no auto-dismiss)
toast.show({
  title: 'Update available',
  duration: Infinity,
});
```

### Position Override

```tsx
// Show at top
toast.show({
  title: 'Top toast',
  position: 'top',
});

// Show at bottom (default)
toast.show({
  title: 'Bottom toast',
  position: 'bottom',
});
```

### Programmatic Control

```tsx
const toast = useToast();

// Get toast ID for later control
const toastId = toast.show({ title: 'Message' });

// Hide specific toast
toast.hide(toastId);

// Hide all toasts
toast.hideAll();

// Update existing toast
toast.update(toastId, { title: 'Updated message' });
```

          <Button onPress={() => toast.show({ title: 'Default toast message' })}>
            Show Default Toast
          </Button>

          <Button color="green" onPress={() => toast.success({ title: 'Operation successful!' })}>
            Show Success Toast
          </Button>

          <Button color="red" onPress={() => toast.error({ title: 'Something went wrong' })}>
            Show Error Toast
          </Button>

          <Button color="yellow" onPress={() => toast.warning({ title: 'Warning: Please review' })}>
            Show Warning Toast
          </Button>

          <Button color="blue" onPress={() => toast.info({ title: 'Here is some information' })}>
            Show Info Toast
          </Button>
        </Flex>
      </Card>

      {/* Toast with Action */}
      <Card>
        <Flex direction="column" gap={12} paddingHorizontal={4}>
          <Heading size={4}>With Action</Heading>

          <Button
            color="red"
            onPress={() =>
              toast.error({
                title: 'File deleted',
                description: 'Your file has been moved to trash.',
                action: {
                  label: 'Undo',
                  onPress: () => toast.success({ title: 'File restored!' }),
                },
                showClose: false,
              })
            }
          >
            Delete File (with Undo)
          </Button>
        </Flex>
      </Card>

      {/* Custom Duration */}
      <Card>
        <Flex direction="column" gap={12} paddingHorizontal={4}>
          <Heading size={4}>Custom Duration</Heading>

          <Button
            onPress={() =>
              toast.show({
                title: 'Quick toast (2s)',
                duration: 2000,
              })
            }
          >
            2 Second Toast
          </Button>

          <Button
            onPress={() =>
              toast.show({
                title: 'Long toast (10s)',
                description: 'This toast will stay for 10 seconds',
                duration: 10000,
              })
            }
          >
            10 Second Toast
          </Button>
        </Flex>
      </Card>

      {/* Position Override */}
      <Card>
        <Flex direction="column" gap={12} paddingHorizontal={4}>
          <Heading size={4}>Position Override</Heading>

          <Button
            onPress={() =>
              toast.show({
                title: 'Top positioned toast',
                position: 'top',
              })
            }
          >
            Show at Top
          </Button>

          <Button
            onPress={() =>
              toast.show({
                title: 'Bottom positioned toast',
                position: 'bottom',
              })
            }
          >
            Show at Bottom
          </Button>
        </Flex>
      </Card>

      {/* Stacking Demo */}
      <Card>
        <Flex direction="column" gap={12} paddingHorizontal={4}>
          <Heading size={4}>Stacking</Heading>
          <Text color="gray" size={2}>
            Multiple toasts stack. Max 3 visible by default.
          </Text>

          <Button
            onPress={() => {
              toast.show({ title: 'First toast' });

              setTimeout(() => toast.show({ title: 'Third toast' }), 600);
              setTimeout(() => toast.success({ title: 'All done!' }), 900);
            }}
          >
            Show Multiple Toasts
          </Button>
        </Flex>
      </Card>

      {/* Programmatic Control */}
      <Card>
        <Flex direction="column" gap={12} paddingHorizontal={4}>
          <Heading size={4}>Programmatic Control</Heading>

          <Flex direction="row" gap={8}>
            <Button color="red" variant="soft" onPress={() => toast.hideAll()}>
              Hide All
            </Button>
          </Flex>
        </Flex>
      </Card>

      {/* Without Close Button */}
      <Card>
        <Flex direction="column" gap={12} paddingHorizontal={4}>
          <Heading size={4}>Without Close Button</Heading>

          <Button
            onPress={() =>
              toast.show({
                title: 'No close button',
                description: 'Swipe to dismiss',
                showClose: false,
              })
            }
          >
            Show Toast (no X)
          </Button>
        </Flex>
      </Card>
    </Flex>
  );
}

export default function ToastDemoScreen() {
  return (
    <ThemeProvider
      mode="light"
      themeOptions={{ accentColor: 'blue', radius: 'large', scaling: 1 }}
      toastConfig={{
        position: 'bottom',
        maxVisible: 3,
        duration: 5000,
        swipeToDismiss: true,
      }}
    >
      <ScrollView>
        <View style={{ flex: 1 }}>
          <ToastDemo />

      </ScrollView>
    </ThemeProvider>
