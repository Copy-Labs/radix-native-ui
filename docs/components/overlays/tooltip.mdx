# Tooltip

`Tooltip` is an overlay component for displaying contextual information on hover or long-press. Supports both simple usage and compound component API for more control.

## Import

```tsx
import { Tooltip } from 'radix-native-ui';
```

## Usage

```tsx
import { Tooltip, IconButton } from 'radix-native-ui';

<Tooltip content="Copy to clipboard">
  <IconButton>
    <Icon name="copy" />
  </IconButton>
</Tooltip>
```

## Components

Tooltip supports both simple usage and compound component API.

### Simple Usage

The simple API combines Root, Trigger, Portal, and Content into one component.

### Tooltip.Root

The root container component that manages tooltip state.

<PropTable>
  <Prop name="children" type="ReactNode">
    Tooltip components.
  </Prop>
  <Prop name="open" type="boolean">
    Controlled open state.
  </Prop>
  <Prop name="onOpenChange" type="(open: boolean) => void">
    Callback when open state changes.
  </Prop>
  <Prop name="defaultOpen" type="boolean" default="false">
    Initial open state.
  </Prop>
  <Prop name="delayDuration" type="number" default="300">
    Delay before showing tooltip (ms).
  </Prop>
  <Prop name="longPressDuration" type="number" default="500">
    Long press duration to show tooltip on mobile (ms).
  </Prop>
</PropTable>

### Tooltip.Trigger

Element that triggers the tooltip.

<PropTable>
  <Prop name="children" type="ReactNode">
    Trigger element.
  </Prop>
  <Prop name="asChild" type="boolean" default="true">
    Merge props onto immediate child element.
  </Prop>
</PropTable>

### Tooltip.Portal

Renders tooltip content in a Modal.

<PropTable>
  <Prop name="children" type="ReactNode">
    Portal content.
  </Prop>
</PropTable>

### Tooltip.Content

The tooltip content container.

<PropTable>
  <Prop name="children" type="ReactNode">
    Tooltip content.
  </Prop>
  <Prop name="side" type="'top' | 'right' | 'bottom' | 'left'" default="'top'">
    Preferred side to render.
  </Prop>
  <Prop name="sideOffset" type="number" default="8">
    Distance from trigger.
  </Prop>
  <Prop name="align" type="'start' | 'center' | 'end'" default="'center'">
    Alignment relative to trigger.
  </Prop>
  <Prop name="alignOffset" type="number" default="0">
    Offset from alignment.
  </Prop>
  <Prop name="avoidCollisions" type="boolean" default="true">
    Avoid viewport collisions.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'">
    Border radius. Inherits from theme by default.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

### Tooltip.Arrow

Optional arrow indicator.

<PropTable>
  <Prop name="size" type="number" default="6">
    Arrow size.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

## Examples

### Basic Tooltip

```tsx
<Tooltip content="This is a tooltip">
  <Button>Hover me</Button>
</Tooltip>
```

### Positioned Tooltips

```tsx
<Flex gap={4}>
  <Tooltip content="Top tooltip" side="top">
    <Button>Top</Button>
  </Tooltip>

  <Tooltip content="Right tooltip" side="right">
    <Button>Right</Button>
  </Tooltip>

  <Tooltip content="Bottom tooltip" side="bottom">
    <Button>Bottom</Button>
  </Tooltip>

  <Tooltip content="Left tooltip" side="left">
    <Button>Left</Button>
  </Tooltip>
</Flex>
```

### Icon Button Tooltips

```tsx
<Flex gap={2}>
  <Tooltip content="Copy">
    <IconButton>
      <Icon name="copy" />
    </IconButton>
  </Tooltip>

  <Tooltip content="Edit">
    <IconButton>
      <Icon name="pencil" />
    </IconButton>
  </Tooltip>

  <Tooltip content="Delete">
    <IconButton color="red">
      <Icon name="trash" />
    </IconButton>
  </Tooltip>
</Flex>
```

### With Delay

```tsx
<Tooltip content="Shows after 1 second" delayDuration={1000}>
  <Button>Delayed tooltip</Button>
</Tooltip>
```

### Rich Content

```tsx
<Tooltip
  content={
    <Flex direction="column" gap={1}>
      <Text weight="medium">Keyboard Shortcut</Text>
      <Text size={1}>Press ⌘ + C to copy</Text>
    </Flex>
  }
>
  <IconButton>
    <Icon name="keyboard" />
  </IconButton>
</Tooltip>
```

### Toolbar

```tsx
<Card>
  <Flex gap={1}>
    <Tooltip content="Bold (⌘B)">
      <IconButton variant="soft">
        <Icon name="bold" />
      </IconButton>
    </Tooltip>

    <Tooltip content="Italic (⌘I)">
      <IconButton variant="soft">
        <Icon name="italic" />
      </IconButton>
    </Tooltip>

    <Tooltip content="Underline (⌘U)">
      <IconButton variant="soft">
        <Icon name="underline" />
      </IconButton>
    </Tooltip>

    <Tooltip content="Link">
      <IconButton variant="soft">
        <Icon name="link" />
      </IconButton>
    </Tooltip>
  </Flex>
</Card>
```

### Form Field Help

```tsx
<Flex direction="column" gap={1}>
  <Flex direction="row" gap={1} align="center">
    <Text size={2} weight="medium">API Key</Text>
    <Tooltip content="Your API key is found in the Developer Settings">
      <IconButton size={1} variant="ghost">
        <Icon name="info" />
      </IconButton>
    </Tooltip>
  </Flex>
  <TextField placeholder="Enter your API key" />
</Flex>
```
